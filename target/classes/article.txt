1) CHECK DATA_PUMP_DIR LOCATION

SELECT * FROM SYS.DBA_DIRECTORIES WHERE DIRECTORY_NAME='DATA_PUMP_DIR';

2) CREATE/REPLACE DIRECTORY LOCATION

CREATE OR REPLACE DIRECTORY DATA_PUMP_DIR as '/disk00/oracle/dpdump/';

3) EXPORT/IMPORT

 - FULL EXPORT

	expdp directory=DATA_PUMP_DIR FULL=Y dumpfile=SAMPLE.dmp logfile=SAMPLE.log compression=all reuse_dumpfiles=y

	impdp directory=DATA_PUMP_DIR FULL=Y dumpfile=SAMPLE.dmp logfile=SAMPLE.log

 - SCHEMA LEVEL EXPORT

	expdp schemas=OSDS directory=DATA_PUMP_DIR dumpfile=SAMPLE.dmp logfile=SAMPLE.log

	impdp schemas=OSDS directory=DATA_PUMP_DIR dumpfile=SAMPLE.dmp logfile=SAMPLE.log

 - TABLE LEVEL EXPORT

	expdp TABLES=ADMIN.PERSON directory=DATA_PUMP_DIR dumpfile=SAMPLE.dmp logfile=SAMPLE.log

 - REMAP TABLESPACE

	impdp full=yes REMAP_TABLESPACE=PROJECTCHARTER_TS:SERVICE_REQUEST_TS directory=DATA_PUMP_DIR dumpfile=SAMPLE.dmp logfile=SAMPLE.log status=10

 - REMAP MULTIPLE TABLESPACE

	impdp REMAP_TABLESPACE=DATA_MEDIUM:USERS REMAP_TABLESPACE=DATA_LARGE:USERS      
	REMAP_TABLESPACE=DATA_HUGE:USERS REMAP_TABLESPACE=DATA_TINY:USERS
	REMAP_TABLESPACE=DATA_PIC:USERS

 - Get reverse engineered DDL

	Impdp directory=DATA_PUMP_DIR dumpfile=SAMPLE.dmp SQLFILE=ReverseDDL.sql
